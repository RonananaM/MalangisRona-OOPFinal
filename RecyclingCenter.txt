class RecyclingCenter:
    def __init__(self, name, location, capacity):
        self.name = name
        self.location = location  # (latitude, longitude)
        self.capacity = capacity  # Maximum material weight in kilograms
        self.current_load = 0  # Current material load
        self.materials = []  # List of RecyclingMaterial objects

    def add_material(self, material):
        if self.current_load + material.weight > self.capacity:
            return f"Cannot add material. {self.name} is at full capacity!"
        self.materials.append(material)
        self.current_load += material.weight
        return f"Added {material.weight} kg of {material.material_type} to {self.name}."

    def process_all_materials(self):
        for material in self.materials:
            print(material.process_material())
        self.materials.clear()
        self.current_load = 0
        return f"All materials processed at {self.name}."
